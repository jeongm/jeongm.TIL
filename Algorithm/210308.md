```c++
		arry1.push_back(x);
	}

	cout<< "가장 많이 등장하는 숫자 : " << majority1(arry1);
}

// 주어진 배열에서 가장 많이 등장하는 숫자를 반환하기
int majority1(const vector<int>& A)
{

	int n = A.size();
	int max = 0, maxa=A[0];

	for (int i = 0; i < n; i++){
		int num=0;
		for (int j = 0; j < n; j++){
			if (A[i] == A[j])
				num++;
		}

		if (num > max)
		{
			max = num;
			maxa = A[i];
		}
	}

	return maxa;
}
```

```c++
#include <iostream>
#include <vector>

using namespace std;

vector<double> movingAverage1(const vector<double>& A);

int main()
{
	vector<double> M;

	int x;

	cout << "M개의 측정치를 입력하시오 : ";
	while (cin >> x) {
		M.push_back(x);
	}

	vector<double> m_avg = movingAverage1(M);

	for (int i = 0; i < M.size(); i++) {
		cout << m_avg[i] << " ";
	}
	
	
}

// 이동 평균 구하기
vector<double> movingAverage1(const vector<double>& A)
{
	int n = A.size();
	vector<double> m_avg;

	m_avg.push_back(0);

	for (int i = 1; i < n; i++) {
		m_avg.push_back((A[i - 1] + A[i])/2);
	}// n번

	return m_avg;
}
```
