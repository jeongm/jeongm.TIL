```c++
#include <iostream>
#include <vector>

using namespace std;

int majority1(const vector<int>& A);

int main()
{
	vector<int> arry1;

	int x;

	cout << "Please enter arry1 : ";
	while (cin >> x) {
		arry1.push_back(x);
	}

	cout<< "가장 많이 등장하는 숫자 : " << majority1(arry1);
}

// 주어진 배열에서 가장 많이 등장하는 숫자를 반환하기
int majority1(const vector<int>& A)
{

	int n = A.size();
	int max = 0, maxa=A[0];

	for (int i = 0; i < n; i++){
		int num=0;
		for (int j = 0; j < n; j++){
			if (A[i] == A[j])
				num++;
		}

		if (num > max)
		{
			max = num;
			maxa = A[i];
		}
	}

	return maxa;
}
```
